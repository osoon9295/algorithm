function solution(numbers, direction) {

    answer = numbers.slice(-1);
    for (let i = 0; i < numbers.length - 1; i++) {
      answer.push(numbers[i]);
    }
  }
  if (direction === "left") {
    for (let i = 0; i < numbers.length - 1; i++) {
      answer.push(numbers[i + 1]);
    }
    answer.push(numbers[0]);
  }
  return answer;
}
